LSB Python extractor (paste & customize)
from PIL import Image

def lsb_extract(filename, n_bits=1, channel=0):
    """Extract LSB steganography from image"""
    img = Image.open(filename).convert('RGB')
    pixels = list(img.getdata())
    channels = ['R','G','B']
    print(f"Extracting LSB from {channels[channel]} channel")
    
    bits = ''
    for pixel in pixels:
        for bit in range(n_bits):
            bits += str((pixel[channel] >> bit) & 1)
    
    # Try to decode as text
    result = []
    for i in range(0, len(bits)-7, 8):
        byte = int(bits[i:i+8], 2)
        if byte == 0: break
        if 32 <= byte <= 126:
            result.append(chr(byte))
    
    msg = ''.join(result)
    print(f"Decoded: {msg[:200]}")
    
    # Search for flag pattern
    import re
    flags = re.findall(r'[a-zA-Z0-9_]+\{[^}]+\}', msg)
    if flags: print(f"🚩 FLAGS: {flags}")
    return msg
